#include<lpc214x.h>
void delay(){
int i,j;
for(i=0;i<5000;i++)
	{
		for(j=0;j<500;j++);
	}
}
int main(){
	PINSEL0=0X00000000;
	IODIR0=0XFFFFFFFF;
	while(1){
		IOCLR0=0XFFFFFFFF;
		delay();
		IOSET0 |=(1<<0);
		delay();
		IOCLR0 |=(1<<0);
		delay();
		IOSET0 |=(1<<1);
		delay();
		IOCLR0 |=(1<<1);
		delay();
		IOSET0 |=(1<<2);
		delay();
		IOCLR0=(1<<2);
		delay();
		IOSET0 |=(1<<3);
		delay();
		IOCLR0 |=(1<<3);
		delay();
		IOSET0 |=(1<<4);
		delay();
		IOCLR0 |=(1<<4);
		delay();
		IOSET0 |=(1<<5);
		delay();
		IOCLR0 |=(1<<5);
		delay();
		IOSET0 |=(1<<6);
		delay();
		IOCLR0 |=(1<<6);
		delay();
		IOSET0 |=(1<<7);
		delay();
		IOCLR0 |=(1<<7);
		delay();
		
	}
}
